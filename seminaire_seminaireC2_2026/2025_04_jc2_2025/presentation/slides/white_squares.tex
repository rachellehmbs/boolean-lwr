\hfill For $\kim$, $U = u_1\cFF \times u_2\cFF$, $V = v_1\cFF \times v_2\cFF$.

\begin{mybox}[cblue]{Characterization of $\alpha\FF^* \times \beta \FF^* \subset Z_F$}
For cyclotomic mappings of order $d$ over $\cFF$, $[\cLL : \cFF] = 2$
\begin{itemize}
  \item[-] $\cF(\cLL) = c\cFF \quad \implies \quad \forall \ \alpha,\  \alpha \cFF^* \times c\cFF^* \subset Z_F$
  \item[-] Otherwise$^{\blue{*}}$, can only happened if $\cFh_\beta(0) = 0$. 
\end{itemize}
\vspace{10pt}
$^{\blue{*}}$ Full characterization in the abstract
\end{mybox}
