\begin{frame}[label=kim-mapping-prop]
\frametitle{Properties of the Kim mapping (1/2)}
\vspace{-10pt}

\recapbox{ $\kim : \clambda \mapsto \clambda^3 + \clambda^{10} + u\clambda^{24}$ is a cyclotomic mapping over $\lightblue{\FF_{2^3}}$ of order $3$}

\begin{mybox}[cblue]{Immediate corollary}
	$\cF$ cyclotomic $\implies$ $\cF(\clambda\cFF) \subset \cF(\clambda)\cFF$
\end{mybox}
\pause

\begin{mybox}[cred]{The Kim mapping case}
$\cphi \mapsto \cphi^3$ is a bijection over $\lightblue{\FF_{2^3}} \quad \quad \implies \ \ \cF(\clambda\cFF) = \cF(\clambda)\cFF$.
\end{mybox}

\pause
\begin{mybox}[cgreen]{The subspace property \hfill [BDMW10]}
$\cF \from \cLL \to \cLL$ satisfies the $\cFF$-subspace property if:
$$ \cF(\clambda\cFF) = \cF(\clambda)\cFF \quad \forall \ \clambda \in \cLL.$$

$\cF(\cphi\clambda) = \cF(\clambda)\cG_{\clambda}(\cphi)$ where $\cG_{\clambda} \from \cFF \to \cFF$ is bijective.
\end{mybox}

\end{frame}
