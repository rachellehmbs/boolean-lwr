\begin{frame}[label=partitions]
\frametitle{Cosets of a subfield}
\centering
\vspace{-10pt}
\begin{mybox}[cgreen]{Partition into cosets}
\begin{itemize}
	\item[-] $\cFF \subset \cLL$ finite fields of characteristic 2.
	\item[-] $\cFF^*$ multiplicative subgroup of $\cLL^*$ $\implies$ $\cLL^* = \bigsqcup\limits_{\cgamma \in \cGamma} \ \cgamma\cFF^*$
\end{itemize}
\end{mybox}
\begin{mybox}[cred]{}
Any $\clambda \in \cLL^* $ can be uniquely written as $\clambda = \cgamma\cphi$ with $\cgamma \in \cGamma, \cphi \in \cFF^*$.
\end{mybox}
